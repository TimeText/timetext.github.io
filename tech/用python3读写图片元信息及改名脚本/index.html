根据图片元信息修改文件名

目的

整理备份照片到本地，将文件名命名为“2018-04-07-01-01-01.jpeg”这种格式，分别代表“年-月-日-时-分-秒”。

脚本可以从图片元数据中提取时间信息，也可以根据文件名自动写入图片元数据。

图片元数据（metadata）是嵌入到图片文件中的一些标签。比较像文件属性，但是种类繁多。常见的几种标准有：

  • EXIF：通常被数码相机在拍摄照片时自动添加，比如相机型号、镜头、曝光、图片尺寸等信息。
  • IPTC：比如图片标题、关键字、说明、作者、版权等信息。
  • XMP：由Adobe公司制定标准，以XML格式保存。用PhotoShop等Adobe公司的软件制作的图片通常会携带这种信息。

import os,shutil
import time
import pyexiv2
from win32file import CreateFile, SetFileTime, GetFileTime, CloseHandle
from win32file import GENERIC_READ, GENERIC_WRITE, OPEN_EXISTING
from pywintypes import Time  # 可以忽视这个 Time 报错（运行程序还是没问题的）

# 修改文件的创建时间、修改时间、访问时间
def modify_file_time(filePath, createTime, modifyTime, accessTime):
    """
    用来修改任意文件的相关时间属性，时间格式：YYYY-MM-DD HH:MM:SS 例如：2019:02:02 00:01:02
    :param filePath: 文件路径名
    :param createTime: 创建时间
    :param modifyTime: 修改时间
    :param accessTime: 访问时间
    :param offset: 时间偏移的秒数,tuple格式，顺序和参数时间对应
    """
    try:
        offset = (0, 1, 2)
        format = "%Y:%m:%d %H:%M:%S"  # 时间格式
        cTime_t = time.localtime(time.mktime(time.strptime(createTime, format)) + offset[0])
        mTime_t = time.localtime(time.mktime(time.strptime(modifyTime, format)) + offset[1])
        aTime_t = time.localtime(time.mktime(time.strptime(accessTime, format)) + offset[2])
        
        fh = CreateFile(filePath, GENERIC_READ | GENERIC_WRITE, 0, None, OPEN_EXISTING, 0, 0)
        createTimes, accessTimes, modifyTimes = GetFileTime(fh)

        createTimes = Time(int(time.mktime(cTime_t)))
        accessTimes = Time(int(time.mktime(aTime_t)))
        modifyTimes = Time(int(time.mktime(mTime_t)))
        
        SetFileTime(fh, createTimes, accessTimes, modifyTimes)
        CloseHandle(fh)
        print(filePath+'的创建、修改、访问时间已经修改为----：'+createTime)
    except:
        print(filePath+'的创建、修改、访问时间已经修改为----：error error error')

# 文件改名前判断是否有重名，并自动规划名称
def plan_rename(old_name,new_name):
    if(old_name==new_name):
        print("新旧文件名一样，无需更改")
        return 0
    # 先将原文件名替换为临时文件名temporary_name
    temporary_name="temporary."+old_name.split('.')[-1]
    os.rename(old_name,temporary_name)

    # 判断是否存在重名文件，并返回修改后的新文件名
    def is_repeat(file_name):
        new_file_name=file_name
        if os.path.isfile(file_name): # isfile判断参数file_nama为文件，用来检查是否存在同名文件
            new_file_name=file_name[:file_name.rfind('.')]+'('+str(n[0])+')'+file_name[file_name.rfind('.'):]
            n[0]+=1
        if os.path.isfile(new_file_name):# 如果还存在，则递归调用is_repeat函数
            new_file_name=is_repeat(file_name)
        return new_file_name

    # 更名
    n=[1]
    new_name=is_repeat(new_name)
    os.rename(temporary_name,new_name)
    return new_name

        print("no exist")

# 1.输出图片的EXIF、IPTC、XMP信息
def pic_all_info(file):
    pic = pyexiv2.Image(file,encoding='GBK')
    exif = pic.read_exif()
    iptc = pic.read_iptc()
    xmp = pic.read_xmp()
    pic.close()

    print("\n----" + file + "的----EXIF信息：")
    for key in exif:
        print(key+':'+exif[key])

    print("\n----" + file + "的----IPTC信息：")
    for key in iptc:
        print(key+':'+iptc[key])
    
    print("\n----" + file + "的----XMP信息：")
    for key in xmp:
        print(key+':'+xmp[key])

# 2.只输出图片中的EXIF关于时间部分
def pic_only_time_info(file):
    pic = pyexiv2.Image(file,encoding='GBK')
    exif_info = pic.read_exif()
    pic.close()

    if('Exif.Image.DateTime' in exif_info):
        print(file+'的DateTime日期和时间:' + exif_info['Exif.Image.DateTime'])
    if('Exif.Photo.DateTimeOriginal' in exif_info):
        print(file+'的Original创建时间:' + exif_info['Exif.Photo.DateTimeOriginal'])
    if('Exif.Photo.DateTimeDigitized' in exif_info):
        print(file+'的Digitized数字化时间:' + exif_info['Exif.Photo.DateTimeDigitized'])

    print(80*"-")

# 3.EXIF中的时间信息统一为<====EXIF中的DateTime
def pic_alltime_as_datetime(file):
    pic = pyexiv2.Image(file,encoding='GBK')
    exif_info = pic.read_exif()

    # 提取DataTime，新建以便写入
    datetime=exif_info['Exif.Image.DateTime']
    newinfo = {
        'Exif.Photo.DateTimeOriginal' : datetime,
        'Exif.Photo.DateTimeDigitized' : datetime
        }
    pic.modify_exif(newinfo)
    pic.close()
    print(file+"的创建时间和数字化时间已经修改为图片EXIF信息中的DateTime----："+datetime)

# 4.EXIF中的时间信息统一为<====文件名所代表的时间
def pic_alltime_as_filename(file):
    # 在文件名中提取年月日时分秒
    newtime=file[0:4]+':'+file[5:7]+':'+file[8:10]+' '+file[11:13]+':'+file[14:16]+':'+file[17:19]

    pic = pyexiv2.Image(file,encoding='GBK')
    exif_info = pic.read_exif()

    newinfo = {
        'Exif.Image.DateTime':newtime,
        'Exif.Photo.DateTimeOriginal' : newtime,
        'Exif.Photo.DateTimeDigitized' : newtime
        }

    pic.modify_exif(newinfo)
    pic.close()
    print(file+"的EXIF信息中的时间已修改为文件名所表示的时间----："+ newtime)

# 5.图片文件名称改为<====图片EXIF中的DateTime
def pic_filename_as_exif_datetime(file):
    pic = pyexiv2.Image(file,encoding='GBK')
    exif_info = pic.read_exif()
    pic.close()


    if('Exif.Image.DateTime' not in exif_info):
        gofile="../图片EXIF中的DateTime 不存在/"+file
        mymovefile(file,gofile)
        return 0

    # 提取DataTime
    datetime=exif_info['Exif.Image.DateTime']

    # 获取后缀
    tail=file.split('.')

    # 提取成新文件名
    new_file_name=datetime[0:4]+'-'+datetime[5:7]+'-'+datetime[8:10]+'-'+datetime[11:13]+'-'+datetime[14:16]+'-'+datetime[17:19]+'.'+tail[-1]
    
    # 有规划的更改文件名
    new_file_name=plan_rename(file,new_file_name)

    print('将文件名改为EXIF中的时间')

# 9.将Unix文件戳命名的文件名改为<====北京时间命名的文件名
def unix_to_bjtime(file):
    # 获取后缀
    tail=file.split('.')

    # 提取时间
    try:
        unix_time=int(file.split('.')[0])
    except Exception as e:
        print("文件",file,"出现错误,请确认UNIX时间戳格式正确。错误信息：",e)
        exit()

    # 检查是不是符合UNIX时间戳长度
    if(len(str(unix_time)) != 10):
        print(file,"的文件名和UNIX时间戳长度不符，请确认UNIX时间戳格式正确。")
        exit()
    
    # 转换
    bjtime=time.gmtime(unix_time)

    # 格式化
    new_file_name=time.strftime("%Y-%m-%d-%H-%M-%S",bjtime)
    new_file_name=new_file_name+'.'+tail[-1]

    # 更名
    new_file_name=plan_rename(file,new_file_name)

    print("UNIX时间戳转换:"+file+"<===="+new_file_name)

根据视频拍摄日期修改文件名

下载：https://exiftool.org/

查看元数据：

查看所有的时间类标签：exiftool -time:all c:\aaa.mp4

查看视频文件所有的元数据。其中-a表示允许提取重复的标签；-s表示简明输出格式；-g1表示按照标签组组织输出；-u表示提取未知标签。exiftool -a -s -u -g1 c:\aaa.mp4

修改文件名：exiftool -overwrite_original -r -api "QuickTimeUTC" '-FileName<CreateDate' -d "%Y-%m-%d-%H-%M-%S%%-c.%%e" *.mp4

-overwrite_original表示覆盖原文件。

-r表示遍历子目录，如果没有子目录可以不使用该参数。

-api表示设置ExifTool的API选项。

QuickTimeUTC表示采用UTC记录时间，如果存在这个选项，CreateDate标签中的时间会自动转换成系统所在时区的时间。QuickTimeUTC仅对QuickTime分类下的标签有用。

-FileName<CreateDate表示用CreateDate标签中的时间来重命名文件。

-d %Y-%m-%d-%H-%M-%S%%-c.%%e表示文件名格式。 其中，-d表示设置格式或者日期/时间的值。 %Y-%m-%d-%H-%M-%S即为视频拍摄日期和时间； %%-c表示如果存在重名文件，则自动添加数字后缀； %%e表示采用源文件的扩展名。
