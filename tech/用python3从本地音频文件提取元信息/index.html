用Python3从本地音频文件提取元信息

 2021.5.11  561

from pymediainfo import MediaInfo
from moviepy.editor import VideoFileClip
import json
import os
import openpyxl


def getinfo(file):
    # 查询信息
    media_info = MediaInfo.parse(file)
    data = media_info.to_data()
    # print(data)

    # 格式化输出字典信息
    # print(json.dumps(data,indent=4,ensure_ascii=False, sort_keys=False,separators=(',', ':')))
    '''
    sort_keys：是否按照字典排序（a-z）输出，True代表是，False代表否。
    indent=4：设置缩进格数，一般由于Linux的习惯，这里会设置为4。
    separators：设置分隔符, 带空格输出 separators=(’, ', ': ‘), 不带空格输出separators=(’, ', ': ')
    ensure_ascii: 是否显示ascii这个码，默认是ture，设置为False可显示中文
    '''

    # print(type(data['tracks'][0]))
    # 定义字典
    info={}

    # 文件名称
    info['filename']=data['tracks'][0]["file_name"] if "file_name" in data['tracks'][0] else None

    # 文件后缀
    info['filetype']=data['tracks'][0]["file_extension"] if "file_extension" in data['tracks'][0] else None

    # 歌曲标题
    info['title']=data['tracks'][0]["title"] if "title" in data['tracks'][0] else None

    # 演唱者
    info['performer']=data['tracks'][0]["performer"] if "performer" in data['tracks'][0] else None

    # 唱片集
    info['album']=data['tracks'][0]["album"] if "album" in data['tracks'][0] else None

    # 持续时间
    info['duration']=data['tracks'][0]["other_duration"][0] if "other_duration" in data['tracks'][0] else None
    
    # 测试输出字典
    # print(info)

    # 返回保存信息的字典
    return info

# 写入excel
def toexcel(info):
    workbook = openpyxl.Workbook()
    sheet=workbook.active

    sheet.append(['标记1','标记2','歌曲名','演唱者','持续时长','唱片集','文件名称','文件后缀'])
    for i in info:
        sheet.append(['','',i['title'],i['performer'],i['duration'],i['album'],i['filename'],i['filetype']])

    workbook.save('info.xlsx')
    print("相关信息已经提取至同目录下的info.xlsx文件中")

# 遍历
def bianli(path):
    # 存储所有数据
    alldata=[]
    for root,dirs,files in os.walk(path):
        for nowfile in files:
            if (nowfile.endswith(('.mp3','.wav'))):
                info=getinfo(nowfile)
                #print(info)
                alldata.append(info)
    # print(alldata[0])
    # 将数据写入excel
    toexcel(alldata)

if __name__ == '__main__':
    bianli(r'./')
